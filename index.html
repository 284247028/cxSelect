【settings 参数说明】（可选）

名称 默认值 说明
url "js/city.js" 列表数据文件路径（josn格式）
select [] 下拉框包含的 className
nodata null 子级无数据时select的状态。可设置为："none"(display隐藏) | "hidden"(visibility隐藏)
required false 是否为必选。设为 flase 时，会在列表头部添加 <option value="0">请选择</option> 选项。

【DOM 结构】

<!--
select 的 class 任意取值，必须放在 id="element_id" 内部，不限层级
如需设置 select 默认值，加上 data-val 属性，例：<select class="province" data-val="浙江"></select>
-->
<div id="element_id">
	<select class="province"></select>
	<select class="city" disabled="disabled"></select>
	<select class="area" disabled="disabled"></select>
</div>

【JavaScript 调用方式】

// selectes 为数组形式，请注意排放顺序
$("#element_id").cxSelect({
	selects:["province","city","area"],
	nodata:"none"
});

// 设置全局默认值，需在引入 <script src="js/jquery.cxselect.js"></script> 之后设置
$.cxSelect.defaults.url="/js/date/city.js";
$.cxSelect.defaults.nodata="none";
